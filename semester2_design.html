2. Sequence Diagrams

Showing interactions between system components over time

📖 What is a Sequence Diagram?

A Sequence Diagram (UML) shows which components talk to each other, in what order, and what messages they exchange — across time (top = earliest, bottom = latest).

ElementMeaning
Lifeline (vertical dashed line)Represents a participant — e.g., Browser, Backend API, Database, Razorpay
Activation bar (rectangle on lifeline)Shows when a component is actively processing
Solid arrow →A call / request being sent
Dashed arrow ⤍A response / return value
Alt / Opt / Loop framesConditional paths, optional blocks, and repeated steps
X on lifelineComponent terminates or an error kills the flow

When to draw: Draw sequence diagrams for every multi-component interaction: user login, order placement, payment, delivery status update. Crucially, always draw both the success path and the error path.

Key Rule: Never Create an Order Before Payment Succeeds

The order row is only inserted after the payment signature is verified as genuine. This prevents "ghost orders" — orders in the DB that were never paid for. Verify first, persist second.

Sequence Diagram 4: Expired JWT — Token Refresh ❌ → ✅

Every protected endpoint must handle an expired token gracefully. The frontend silently refreshes the token and retries — the user never sees the error.

BaseUser.js — Abstract Domain Class

// src/models/domain/BaseUser.js
const bcrypt = require('bcryptjs');
const { prisma } = require('../../lib/prisma');
const { generateTokens } = require('../../lib/jwt');

class BaseUser {
  constructor({ id, name, email, phone, role, isActive = true, createdAt }) {
    if (new.target === BaseUser) {
      throw new Error('BaseUser is abstract — use Customer, Owner, Partner or Admin');
    }
    this.id = id;
    this.name = name;
    this.email = email;
    this.phone = phone;
    this.role = role;
    this.isActive = isActive;
    this.createdAt = createdAt;
  }

  // Concrete — identical logic for every role
  static async login(email, password) {
    const user = await prisma.user.findUnique({ where: { email } });
    if (!user) throw Object.assign(new Error('INVALID_CREDENTIALS'), { status: 401 });

    const match = await bcrypt.compare(password, user.passwordHash);
    if (!match) throw Object.assign(new Error('INVALID_CREDENTIALS'), { status: 401 });

    if (!user.isActive) throw Object.assign(new Error('ACCOUNT_SUSPENDED'), { status: 403 });

    await prisma.user.update({ where: { id: user.id }, data: { lastLogin: new Date() } });
    return generateTokens(user);   // returns { accessToken, refreshToken }
  }

  // Abstract — subclass MUST override this
  getProfile() {
    throw new Error(`${this.constructor.name} must implement getProfile()`);
  }
}

module.exports = { BaseUser };

Customer.js — Subclass with Overrides

// src/models/domain/Customer.js
const { BaseUser } = require('./BaseUser');
const { prisma } = require('../../lib/prisma');

class Customer extends BaseUser {
  constructor(data) {
    super({ ...data, role: 'CUSTOMER' });
    this.savedAddresses = data.addresses ?? [];
    this.orderHistory   = data.orders ?? [];
  }

  // Override abstract method — returns Customer-specific profile
  getProfile() {
    return {
      id:        this.id,
      name:      this.name,
      email:     this.email,
      phone:     this.phone,
      addresses: this.savedAddresses,
    };
  }

  // Customer-only method — maps to OrderService
  async placeOrder({ restaurantId, items, deliveryAddrId, paymentMethod }) {
    const { OrderService } = require('../services/order.service');
    return OrderService.placeOrder({
      customerId: this.id,
      restaurantId,
      items,
      deliveryAddrId,
      paymentMethod,
    });
  }

  async getOrderHistory() {
    return prisma.order.findMany({
      where: { customerId: this.id },
      orderBy: { createdAt: 'desc' },
      include: { items: { include: { menuItem: true } }, payment: true },
    });
  }
}

module.exports = { Customer };

OrderService.js — The Service Layer

// src/services/order.service.js
const { prisma } = require('../../lib/prisma');

const VALID_TRANSITIONS = {
  PLACED:    ['ACCEPTED', 'CANCELLED'],
  ACCEPTED:  ['PREPARING', 'CANCELLED'],
  PREPARING: ['PICKED_UP'],
  PICKED_UP: ['DELIVERED'],
  DELIVERED: [],   // terminal state
  CANCELLED: [],   // terminal state
};

const OrderService = {

  async placeOrder({ customerId, restaurantId, items, deliveryAddrId, paymentMethod }) {
    // 1. Validate items + fetch current prices from DB (never trust client prices)
    const menuItems = await prisma.menuItem.findMany({
      where: { id: { in: items.map(i => i.menuItemId) }, isAvailable: true },
    });
    if (menuItems.length !== items.length) throw new Error('ITEM_UNAVAILABLE');

    // 2. Calculate totals
    const subtotal = items.reduce((sum, i) => {
      const mi = menuItems.find(m => m.id === i.menuItemId);
      return sum + mi.price * i.quantity;
    }, 0);
    const deliveryFee  = 40.00;
    const totalAmount  = subtotal + deliveryFee;

    // 3. Create order + line items in a single transaction
    const order = await prisma.$transaction(async (tx) => {
      const o = await tx.order.create({
        data: {
          customerId, restaurantId, deliveryAddrId, paymentMethod,
          subtotal, deliveryFee, totalAmount, status: 'PLACED',
          items: {
            create: items.map(i => ({
              menuItemId: i.menuItemId,
              quantity:   i.quantity,
              unitPrice:  menuItems.find(m => m.id === i.menuItemId).price,
              subtotal:   menuItems.find(m => m.id === i.menuItemId).price * i.quantity,
            })),
          },
        },
        include: { items: true },
      });
      return o;
    });

    return order;
  },

  async transitionStatus(orderId, newStatus, actorId) {
    const order = await prisma.order.findUniqueOrThrow({ where: { id: orderId } });
    const allowed = VALID_TRANSITIONS[order.status];
    if (!allowed.includes(newStatus)) {
      throw Object.assign(
        new Error(`Cannot move order from ${order.status} to ${newStatus}`),
        { status: 400 }
      );
    }
    return prisma.order.update({
      where: { id: orderId },
      data:  { status: newStatus, updatedAt: new Date() },
    });
  },
};

module.exports = { OrderService };
Key Engineering Decisions in the Code Above
DecisionWhy it matters
Prices fetched from DB — not trusted from clientPrevents price manipulation attacks ("I'll pay ₹1 for a ₹500 item")
Prisma $transaction for order + itemsIf any item insert fails, the whole order is rolled back — no partial orders
VALID_TRANSITIONS map in service layerFinite State Machine enforced in code — invalid status changes rejected before hitting DB
Abstract getProfile() throws if not overriddenFail fast at runtime — better than silently returning wrong data

REST API Design Principles for Food Delivery

PrincipleExample
Use nouns, not verbs in URLs✅ GET /orders/   ❌ GET /getOrders/
Use plural nouns✅ /restaurants/   ❌ /restaurant/
Nest resources sensibly✅ GET /restaurants/7/menu/
Use HTTP verbs correctlyGET=read, POST=create, PUT/PATCH=update, DELETE=remove

Version your API/api/v1/orders/ — so future breaking changes go to /api/v2/

Full SQL DDL — Production Schema (PostgreSQL)

-- ============================================================
-- ONLINE FOOD DELIVERY SYSTEM — Physical DB Schema v2.0
-- Engine: PostgreSQL 15+   ORM: Prisma
-- ============================================================

-- 1. USERS (single table — all roles via role column)
CREATE TABLE users (
    user_id       SERIAL PRIMARY KEY,
    name          VARCHAR(100)  NOT NULL,
    email         VARCHAR(255)  UNIQUE NOT NULL,
    password_hash VARCHAR(255)  NOT NULL,
    phone         VARCHAR(15),
    role          VARCHAR(20)   NOT NULL
                  CHECK (role IN ('CUSTOMER','OWNER','PARTNER','ADMIN')),
    is_active     BOOLEAN       NOT NULL DEFAULT TRUE,
    last_login    TIMESTAMP,
    created_at    TIMESTAMP     NOT NULL DEFAULT NOW()
);
CREATE INDEX idx_users_email   ON users (email);        -- fast login lookup
CREATE INDEX idx_users_role    ON users (role);         -- filter by role

-- 2. ADDRESSES
CREATE TABLE addresses (
    address_id  SERIAL PRIMARY KEY,
    user_id     INT           NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    label       VARCHAR(50),                            -- "Home", "Office"
    street      VARCHAR(255)  NOT NULL,
    city        VARCHAR(100)  NOT NULL,
    pincode     VARCHAR(10),
    lat         DECIMAL(9,6),
    lng         DECIMAL(9,6),
    is_default  BOOLEAN       NOT NULL DEFAULT FALSE,
    created_at  TIMESTAMP     NOT NULL DEFAULT NOW()
);
CREATE INDEX idx_addresses_user ON addresses (user_id);

-- 3. RESTAURANTS
CREATE TABLE restaurants (
    restaurant_id  SERIAL PRIMARY KEY,
    owner_id       INT           NOT NULL REFERENCES users(user_id),
    name           VARCHAR(150)  NOT NULL,
    cuisine_type   VARCHAR(50),
    address        TEXT,
    status         VARCHAR(20)   NOT NULL DEFAULT 'PENDING'
                   CHECK (status IN ('PENDING','APPROVED','REJECTED','SUSPENDED')),
    avg_rating     DECIMAL(3,2)  NOT NULL DEFAULT 0.00,
    is_open        BOOLEAN       NOT NULL DEFAULT FALSE,
    created_at     TIMESTAMP     NOT NULL DEFAULT NOW()
);
CREATE INDEX idx_restaurants_status  ON restaurants (status);
CREATE INDEX idx_restaurants_cuisine ON restaurants (cuisine_type);
CREATE INDEX idx_restaurants_owner   ON restaurants (owner_id);

-- 4. MENU ITEMS
CREATE TABLE menu_items (
    item_id        SERIAL PRIMARY KEY,
    restaurant_id  INT           NOT NULL REFERENCES restaurants(restaurant_id) ON DELETE CASCADE,
    name           VARCHAR(150)  NOT NULL,
    description    TEXT,
    price          DECIMAL(8,2)  NOT NULL CHECK (price >= 0),
    category       VARCHAR(50),
    is_veg         BOOLEAN       NOT NULL DEFAULT TRUE,
    is_available   BOOLEAN       NOT NULL DEFAULT TRUE,
    image_url      VARCHAR(500),
    created_at     TIMESTAMP     NOT NULL DEFAULT NOW()
);
CREATE INDEX idx_menu_restaurant ON menu_items (restaurant_id, is_available);

-- 5. ORDERS
CREATE TABLE orders (
    order_id         SERIAL PRIMARY KEY,
    customer_id      INT           NOT NULL REFERENCES users(user_id),
    restaurant_id    INT           NOT NULL REFERENCES restaurants(restaurant_id),
    partner_id       INT           REFERENCES users(user_id),   -- NULL until assigned
    delivery_addr_id INT           NOT NULL REFERENCES addresses(address_id),
    status           VARCHAR(20)   NOT NULL DEFAULT 'PLACED'
                     CHECK (status IN ('PLACED','ACCEPTED','PREPARING',
                                       'PICKED_UP','DELIVERED','CANCELLED')),
    subtotal         DECIMAL(10,2) NOT NULL,
    delivery_fee     DECIMAL(6,2)  NOT NULL DEFAULT 40.00,
    total_amount     DECIMAL(10,2) NOT NULL,
    notes            TEXT,
    created_at       TIMESTAMP     NOT NULL DEFAULT NOW(),
    updated_at       TIMESTAMP     NOT NULL DEFAULT NOW()
);
CREATE INDEX idx_orders_customer  ON orders (customer_id, created_at DESC);
CREATE INDEX idx_orders_restaurant ON orders (restaurant_id, status);
CREATE INDEX idx_orders_partner   ON orders (partner_id) WHERE partner_id IS NOT NULL;
CREATE INDEX idx_orders_status    ON orders (status);

-- 6. ORDER ITEMS
CREATE TABLE order_items (
    order_item_id  SERIAL PRIMARY KEY,
    order_id       INT           NOT NULL REFERENCES orders(order_id) ON DELETE CASCADE,
    menu_item_id   INT           NOT NULL REFERENCES menu_items(item_id),
    quantity       INT           NOT NULL CHECK (quantity > 0),
    unit_price     DECIMAL(8,2)  NOT NULL,              -- snapshot at order time
    subtotal       DECIMAL(10,2) NOT NULL
);
CREATE INDEX idx_order_items_order ON order_items (order_id);

-- 7. PAYMENTS
CREATE TABLE payments (
    payment_id      SERIAL PRIMARY KEY,
    order_id        INT           UNIQUE NOT NULL REFERENCES orders(order_id),
    gateway_txn_id  VARCHAR(100),                       -- Razorpay payment_id
    amount          DECIMAL(10,2) NOT NULL,
    method          VARCHAR(20)   CHECK (method IN ('ONLINE','COD')),
    status          VARCHAR(20)   NOT NULL DEFAULT 'PENDING'
                    CHECK (status IN ('PENDING','PAID','FAILED','REFUNDED')),
    paid_at         TIMESTAMP,
    refunded_at     TIMESTAMP
);

-- 8. REVIEWS
CREATE TABLE reviews (
    review_id      SERIAL PRIMARY KEY,
    customer_id    INT  NOT NULL REFERENCES users(user_id),
    restaurant_id  INT  NOT NULL REFERENCES restaurants(restaurant_id),
    order_id       INT  NOT NULL REFERENCES orders(order_id),
    rating         INT  NOT NULL CHECK (rating BETWEEN 1 AND 5),
    comment        TEXT,
    created_at     TIMESTAMP NOT NULL DEFAULT NOW(),
    UNIQUE (customer_id, order_id)                     -- one review per order
);
CREATE INDEX idx_reviews_restaurant ON reviews (restaurant_id);

Critical Indexes — Explained

IndexQuery it speeds upWithout it
idx_users_emailWHERE email = $1 on every loginFull table scan on every login attempt
idx_orders_customer (DESC)Customer's order history, newest firstSeq scan + sort — slow for users with 50+ orders
idx_orders_restaurantRestaurant dashboard — pending ordersFull orders scan to find one restaurant's orders
idx_menu_restaurantMenu page load — available items onlyLoads all items then filters in app memory
idx_reviews_restaurantCalculate avg_rating for a restaurantFull reviews table scan to aggregate one restaurant
idx_orders_partner (partial)Partner's active orders — only where assignedScans all orders including unassigned ones
Over-indexing Warning

Every index speeds up reads but adds overhead to every INSERT / UPDATE. The indexes above target the most frequent read queries. Do not index every column — profile first, index second.

Prisma Schema (ORM layer over the SQL above)

// prisma/schema.prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id           Int       @id @default(autoincrement()) @map("user_id")
  name         String    @db.VarChar(100)
  email        String    @unique @db.VarChar(255)
  passwordHash String    @map("password_hash") @db.VarChar(255)
  phone        String?   @db.VarChar(15)
  role         Role
  isActive     Boolean   @default(true) @map("is_active")
  lastLogin    DateTime? @map("last_login")
  createdAt    DateTime  @default(now()) @map("created_at")

  addresses    Address[]
  restaurants  Restaurant[]
  orders       Order[]       @relation("CustomerOrders")
  deliveries   Order[]       @relation("PartnerOrders")
  reviews      Review[]

  @@index([email])
  @@index([role])
  @@map("users")
}

enum Role {
  CUSTOMER
  OWNER
  PARTNER
  ADMIN
}

model Order {
  id             Int       @id @default(autoincrement()) @map("order_id")
  customer       User      @relation("CustomerOrders", fields: [customerId], references: [id])
  customerId     Int       @map("customer_id")
  restaurant     Restaurant @relation(fields: [restaurantId], references: [id])
  restaurantId   Int       @map("restaurant_id")
  partner        User?     @relation("PartnerOrders", fields: [partnerId], references: [id])
  partnerId      Int?      @map("partner_id")
  deliveryAddr   Address   @relation(fields: [deliveryAddrId], references: [id])
  deliveryAddrId Int       @map("delivery_addr_id")
  status         OrderStatus @default(PLACED)
  subtotal       Decimal   @db.Decimal(10, 2)
  deliveryFee    Decimal   @default(40.00) @map("delivery_fee") @db.Decimal(6, 2)
  totalAmount    Decimal   @map("total_amount") @db.Decimal(10, 2)
  createdAt      DateTime  @default(now()) @map("created_at")
  updatedAt      DateTime  @updatedAt @map("updated_at")

  items    OrderItem[]
  payment  Payment?
  review   Review?

  @@index([customerId, createdAt(sort: Desc)])
  @@index([restaurantId, status])
  @@map("orders")
}

enum OrderStatus {
  PLACED
  ACCEPTED
  PREPARING
  PICKED_UP
  DELIVERED
  CANCELLED

6. Wireframes / UI Design

Sketching the user interface before writing frontend code

Wireframes are low-fidelity sketches of each page — they show layout and navigation without worrying about colours or fonts. Create wireframes for every major screen before the frontend developer starts coding.

Key Screens to Wireframe

  • Customer: Home/Search, Restaurant Listing, Restaurant Menu, Cart, Checkout, Order Tracking, Order History, Profile.
  • Restaurant Owner: Dashboard, Incoming Orders, Menu Management (add/edit/delete item).
  • Delivery Partner: Available Orders, Active Order (with status update buttons).
  • Admin: Pending Restaurant Approvals, All Users list, Platform Analytics.
Wireframing Tools

Figma (free for students) is the industry standard. Excalidraw (free, browser-based) is great for quick hand-drawn style sketches. Even paper wireframes scanned as images in your TDD are acceptable at student level.
