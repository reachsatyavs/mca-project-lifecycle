2. Integration Layer

Connecting frontend to backend and all external services

Integration means ensuring that all parts of the system work together correctly — the frontend calls the right API endpoints, error responses from the backend are shown properly on the frontend, and all external services (payment, email, image storage) work in the integrated system.

Frontend API Client (React / Axios)

// src/api/client.js import axios from 'axios'; const apiClient = axios.create({ baseURL: import.meta.env.VITE_API_BASE_URL, // e.g. https://food-api.render.com headers: { 'Content-Type': 'application/json' } }); // Attach JWT token to every request automatically apiClient.interceptors.request.use(config => { const token = localStorage.getItem('accessToken'); if (token) config.headers.Authorization = `Bearer ${token}`; return config; }); // Handle 401 globally — redirect to login apiClient.interceptors.response.use( response => response, error => { if (error.response?.status === 401) { localStorage.removeItem('accessToken'); window.location.href = '/login'; } return Promise.reject(error); } ); export default apiClient;

Definition of Done for every feature

Definition of Done Checklist

  • Feature is implemented and matches all acceptance criteria in the User Story.
  • Unit tests written and passing (coverage maintained at ≥ 70%).
  • Integration test written for the main success and one failure path.
  • Code reviewed and approved by one other team member.
  • PR merged to develop branch (not sitting in a local branch).
  • Feature deployed to staging and manually smoke-tested.
  • Any new API endpoints documented in Swagger.
  • No P1 or P2 bugs open against this feature.

4. Refactoring

⭐ Good to Have

Improving code quality without changing external behaviour

⭐ Good to Have — Improves maintainability and code review feedback. Do this if all mandatory features are complete and time permits.

Refactoring is changing the internal structure of code to make it cleaner, more efficient, or more maintainable — without adding new features or fixing bugs. In Semester 3, you refactor code written quickly in Semester 2.

Common Refactoring Targets in Food Delivery

ProblemRefactoring
Order placement logic is all in the controller (fat controller)Extract to an OrderService class — controller just calls service methods
Database queries duplicated across controllersExtract to a RestaurantRepository module using Prisma
Magic numbers scattered in code (e.g., if (total > 500))Move to named constants: const FREE_DELIVERY_THRESHOLD = 500;
N+1 query in restaurant listingUse Prisma include: { menuItems: true } in one query
Password reset logic duplicated in two routesExtract to PasswordResetService

Refactoring Example — Fat Controller → Service Layer

// BEFORE: All logic in controller (hard to test) // src/controllers/order.controller.js async function create(req, res) { const { restaurant_id, items, delivery_address_id } = req.body; const restaurant = await prisma.restaurant.findUnique({ where: { id: restaurant_id } }); if (!restaurant) return res.status(404).json({ error: 'Restaurant not found' }); let total = 0; for (const item of items) { const menuItem = await prisma.menuItem.findUnique({ where: { id: item.menu_item_id } }); total += menuItem.price * item.quantity; } if (total < restaurant.minimum_order) return res.status(400).json({ error: 'Below minimum' }); const order = await prisma.order.create({ data: { ... } }); res.status(201).json({ data: order }); } // AFTER: Controller delegates to service (easy to unit-test) // src/services/order.service.js class OrderService { async placeOrder({ customerId, restaurantId, items, addressId }) { const restaurant = await this._findRestaurantOrThrow(restaurantId); const { lineItems, total } = await this._calculateTotal(items); this._assertMinimumOrder(total, restaurant.minimum_order); return prisma.order.create({ data: { customerId, restaurantId, total, items: { create: lineItems } } }); } // ... private methods } // src/controllers/order.controller.js (now thin) const orderService = new OrderService(); async function create(req, res) { const order = await orderService.placeOrder({ customerId: req.user.id, ...req.body }); res.status(201).json({ data: order }); }
Golden Rule of Refactoring

Only refactor code that has unit tests. If you refactor without tests, you might silently break behaviour. The test suite is your safety net — it tells you instantly if a refactor broke something.

5. Real-Time Updates (WebSocket / Socket.io)

⭐ Good to Have

Push live order status updates without polling

⭐ Good to Have — Real-time order tracking is a "wow" feature for evaluators. If time is short, polling every 15 s is a perfectly acceptable fallback.

Instead of the frontend calling the API every 15 seconds to check order status (polling), WebSocket pushes updates from server to browser the instant the status changes. This is how real food delivery apps work.

Polling vs WebSocket Comparison

Polling (every 15 s)WebSocket (Socket.io)
ComplexitySimple — just a setInterval + fetchModerate — needs socket server + client
Server loadHigh — 4 req/min per userLow — 1 persistent connection per user
User experienceUp to 15 s delayInstant (< 1 s)
Recommended forSimpler projects, limited timeImpressive demo, final submission polish

Backend: Socket.io Setup (Express)

// src/server.js const http = require('http'); const { Server } = require('socket.io'); const app = require('./app'); const httpServer = http.createServer(app); const io = new Server(httpServer, { cors: { origin: process.env.FRONTEND_URL, credentials: true } }); // Share io instance with the rest of the app app.set('io', io); io.on('connection', (socket) => { console.log(`Client connected: ${socket.id}`); // Customer joins a room named after their order ID socket.on('track_order', (orderId) => { socket.join(`order:${orderId}`); console.log(`${socket.id} tracking order ${orderId}`); }); socket.on('disconnect', () => { console.log(`Client disconnected: ${socket.id}`); }); }); httpServer.listen(process.env.PORT || 3000);

Backend: Emit Status Update When Order Changes

// src/controllers/order.controller.js async function updateStatus(req, res) { const { id } = req.params; const { status } = req.body; const order = await prisma.order.update({ where: { id: parseInt(id) }, data: { status } }); // Push real-time update to all clients tracking this order const io = req.app.get('io'); io.to(`order:${id}`).emit('order_status_updated', { orderId: id, status, updatedAt: new Date().toISOString() }); res.json({ data: order }); }

Frontend: Listen for Updates (React)

// src/pages/OrderTracking.jsx import { useEffect, useState } from 'react'; import { io } from 'socket.io-client'; export function OrderTracking({ orderId }) { const [status, setStatus] = useState('Confirmed'); useEffect(() => { const socket = io(import.meta.env.VITE_API_BASE_URL); // Tell the server which order to track socket.emit('track_order', orderId); // Listen for status updates socket.on('order_status_updated', ({ status: newStatus }) => { setStatus(newStatus); }); // Cleanup on unmount return () => socket.disconnect(); }, [orderId]); return <OrderTimeline currentStatus={status} />; } // Install: npm install socket.io-client

Step 3 — Tool Definitions

Each tool has a Ollama-compatible JSON Schema definition (so the LLM knows what the tool accepts) and an execute function that queries the real database.

// src/ai/tools/getOrderStatus.js
const { prisma } = require('../../lib/prisma');

const getOrderStatusTool = {
  definition: {
    type: 'function',
    function: {
      name: 'get_order_status',
      description: 'Get the current status, delivery estimate, and details of a customer order.',
      parameters: {
        type: 'object',
        properties: {
          orderId: { type: 'number', description: 'The order ID to look up' }
        },
        required: ['orderId']
      }
    }
  },

  async execute({ orderId }) {
    const order = await prisma.order.findUnique({
      where: { id: orderId },
      include: {
        orderItems: { include: { menuItem: true } },
        deliveryPartner: { select: { name: true } },
        restaurant: { select: { name: true } }
      }
    });
    if (!order) return { error: `Order ${orderId} not found` };
    const lateBy = order.deliveredAt && order.estimatedDeliveryAt
      ? Math.round((order.deliveredAt - order.estimatedDeliveryAt) / 60000)
      : null;
    return {
      orderId: order.id,
      status: order.status,
      restaurant: order.restaurant.name,
      items: order.orderItems.map(i => ({ name: i.menuItem.name, qty: i.quantity, price: i.price })),
      totalAmount: order.totalAmount,
      deliveryPartner: order.deliveryPartner?.name,
      lateByMinutes: lateBy
    };
  }
};

module.exports = { getOrderStatusTool };

const resolveIssueTool = {
  definition: {

      name: 'resolve_issue',

Step 4 — FoodBotAgent: The ReAct Loop

// src/ai/FoodBotAgent.js
const Ollama = require('ollama').default;
const { prisma } = require('../lib/prisma');
const { getOrderStatusTool }  = require('./tools/getOrderStatus');
const { resolveIssueTool }    = require('./tools/resolveIssue');
const { recommendDishesTool } = require('./tools/recommendDishes');
const { ownerAnalyticsTool }  = require('./tools/ownerAnalytics');
const { escalateToHumanTool } = require('./tools/escalateToHuman');

const TOOLS   = [getOrderStatusTool, resolveIssueTool, recommendDishesTool, ownerAnalyticsTool, escalateToHumanTool];
const TOOL_MAP = Object.fromEntries(TOOLS.map(t => [t.definition.function.name, t]));

const SYSTEM_PROMPT = `You are FoodBot, a helpful customer support agent for an online food delivery platform.
You have access to tools to check orders, resolve issues, recommend food, and analyse restaurant data.
Always be empathetic, concise, and action-oriented. If you use a tool, wait for its result before responding.
Never make up order details — always use the tools to get real data.`;

class FoodBotAgent {
  constructor() {
    this.ollama = new Ollama({ host: 'http://localhost:11434' });
    this.model  = 'llama3.1:8b';
  }

  async chat(sessionId, userId, userMessage) {
    await prisma.aiMessage.create({ data: { sessionId, role: 'user', content: userMessage } });

    const history = await prisma.aiMessage.findMany({
      where: { sessionId },
      orderBy: { createdAt: 'asc' },
      take: 20
    });

    const messages = [
      { role: 'system', content: SYSTEM_PROMPT },
      ...history.map(m => ({ role: m.role === 'tool' ? 'tool' : m.role, content: m.content }))
    ];

    const finalResponse = await this.runReActLoop(messages, sessionId);

    await prisma.aiMessage.create({ data: { sessionId, role: 'assistant', content: finalResponse } });
    return finalResponse;
  }

  async runReActLoop(messages, sessionId) {
    const toolDefs = TOOLS.map(t => t.definition);

    for (let i = 0; i < 10; i++) {
      const { message } = await this.ollama.chat({ model: this.model, messages, tools: toolDefs });

      if (!message.tool_calls || message.tool_calls.length === 0) return message.content;

      for (const toolCall of message.tool_calls) {
        const { name, arguments: args } = toolCall.function;
        let result;
        try { result = TOOL_MAP[name] ? await TOOL_MAP[name].execute(args) : { error: 'Unknown tool' }; }
        catch (e) { result = { error: e.message }; }

        const resultStr = JSON.stringify(result);
        await prisma.aiMessage.create({
          data: { sessionId, role: 'tool', content: resultStr, toolName: name, toolResult: resultStr }
        });
        messages.push({ role: 'assistant', content: '', tool_calls: [toolCall] });
        messages.push({ role: 'tool', content: resultStr });
      }
    }
    return 'I was unable to complete your request. Please contact support.';
  }
}

module.exports = { FoodBotAgent };

Step 5 — Chat API Endpoint

// src/routes/ai.js
const express = require('express');
const router  = express.Router();
const { prisma }        = require('../lib/prisma');
const { FoodBotAgent }  = require('../ai/FoodBotAgent');
const { authenticateToken } = require('../middleware/auth');

const agent = new FoodBotAgent();

router.post('/chat', authenticateToken, async (req, res) => {
  try {
    const { message, sessionId } = req.body;
    if (!message?.trim()) return res.status(400).json({ error: 'Message cannot be empty' });

    let session;
    if (sessionId) {
      session = await prisma.aiConversationSession.findUnique({ where: { id: sessionId } });
      if (!session || session.customerId !== req.user.id)
        return res.status(403).json({ error: 'Session not found or unauthorised' });
    } else {
      session = await prisma.aiConversationSession.create({
        data: { customerId: req.user.id, status: 'active' }
      });
    }

    const reply = await agent.chat(session.id, req.user.id, message);
    res.json({ sessionId: session.id, message: reply });
  } catch (err) {
    console.error('[AI Chat Error]', err);
    res.status(500).json({ error: 'AI service unavailable. Please try again.' });
  }
});
